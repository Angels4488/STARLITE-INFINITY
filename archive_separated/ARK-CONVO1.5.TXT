        self.simulator = Aer.get_backend('statevector_simulator')
        print("QLE: Qiskit Aer simulator is online.")

    def apply_cognitive_superposition(self, curiosity_level: float) -> QuantumCircuit:
        """

        # Ensure curiosity is between 0 and 1
        curiosity_level = np.clip(curiosity_level, 0, 1)

        num_qubits_to_superpose = int(round(self.num_qubits * curiosity_level))

class AuditoryCortex:
    """
    Processes raw auditory input to extract speech, emotional tone, and
    significant sound patterns.
    """
    def __init__(self):
        """
        Initializes the speech recognition engine.
        """
        print("AUDITORY_CORTEX: Initializing speech recognizer...")
        self.recognizer = sr.Recognizer()
        print("AUDITORY_CORTEX: Ready for processing.")

    def extract_emotional_tone(self, audio_data: np.ndarray, sample_rate: int) -> Dict[str, float]:
        """
        Performs a basic analysis of pitch and energy to infer emotional tone.
        NOTE: True emotion detection is far more complex and would require a dedicated ML model.
        """
        # Pitch estimation using yin algorithm
        pitch, _, _ = librosa.pyin(audio_data, fmin=librosa.note_to_hz('C2'), fmax=librosa.note_to_hz('C7'))
        avg_pitch = np.nanmean(pitch) if not np.all(np.isnan(pitch)) else 0.0
        
        # Energy calculation (Root Mean Square)
        rms = librosa.feature.rms(y=audio_data)
        avg_energy = np.mean(rms)

        return {
            "average_pitch_hz": float(avg_pitch),
            "average_energy": float(avg_energy)
        }

    def process_raw_sound(self, packet: AudioPacket) -> Dict[str, Any]:
        """
        Processes an audio packet to perform speech-to-text and tone analysis.

        Args:
            packet: The AudioPacket containing the waveform and sample rate.

        Returns:
            A dictionary containing the transcribed text and audio features.
        """
        print(f"AUDITORY_CORTEX: Processing auditory data @ {packet.timestamp}")
        audio_data = packet.data
        sample_rate = packet.sample_rate
        
        # Convert our numpy array data into the format SpeechRecognition understands
        audio_for_sr = sr.AudioData(audio_data.tobytes(), sample_rate, sample_width=audio_data.dtype.itemsize)

        analysis_result = {
            "timestamp": packet.timestamp,
            "transcribed_text": None,
            "detected_tone": None,
            "error": None
        }

        try:
            # 1. Perform Speech-to-Text
            # This uses Google's online service by default. Can be configured for offline models.
            print("AUDITORY_CORTEX: Transcribing speech...")
            text = self.recognizer.recognize_google(audio_for_sr)
            analysis_result["transcribed_text"] = text
            print(f"AUDITORY_CORTEX: Transcribed: '{text}'")

            # 2. Perform Emotional Tone Analysis on the raw audio
            print("AUDITORY_CORTEX: Analyzing emotional tone...")
            tone_features = self.extract_emotional_tone(audio_data, sample_rate)
            analysis_result["detected_tone"] = tone_features

        except sr.UnknownValueError:
            # This error means the library couldn't understand the audio (e.g., it's just noise)
            msg = "Could not understand audio (non-speech or unclear)."
            print(f"AUDITORY_CORTEX: {msg}")
            analysis_result["error"] = msg
        except sr.RequestError as e:
            # This error means there was a problem with the speech recognition service
            msg = f"Could not request results from service; {e}"
            print(f"AUDITORY_CORTEX: {msg}")
            analysis_result["error"] = msg
        except Exception as e:
            msg = f"An unexpected error occurred during audio processing: {e}"
            print(f"AUDITORY_CORTEX: {msg}")
            analysis_result["error"] = msg

class VisualCortex:
    """
    Processes raw visual input using a pre-trained CV model to extract
    symbolic representations of objects in the environment.
    """
    def __init__(self, model_weights_path: str, model_config_path: str, labels_path: str):
        """
        Initializes the vision system by loading a pre-trained neural network.
        This is done once to avoid slow load times on every processing call.
        """
        print("VISUAL_CORTEX: Initializing and loading detection model...")
        try:
            self.net = cv2.dnn.readNet(model_weights_path, model_config_path)
            with open(labels_path, 'r') as f:
                self.labels = [line.strip() for line in f.readlines()]
            # Get the output layer names for the forward pass
            layer_names = self.net.getLayerNames()
            self.output_layers = [layer_names[i - 1] for i in self.net.getUnconnectedOutLayers().flatten()]
            print("VISUAL_CORTEX: Model loaded successfully. Ready for processing.")
        except cv2.error as e:
            print(f"FATAL ERROR: Could not load model. OpenCV error: {e}")
            # Exit or raise a critical exception if the model is essential
            raise RuntimeError("VisualCortex failed to initialize.") from e

    def process_raw_vision(self, packet: SensoryPacket, confidence_threshold: float = 0.5) -> List[Dict[str, Any]]:
        """
        Takes a sensory packet, processes the image data through the neural network,
        and returns a list of detected objects (symbols).

        Args:
            packet: The SensoryPacket containing the image and timestamp.
            confidence_threshold: The minimum probability to filter weak detections.

        Returns:
            A list of dictionaries, where each dictionary represents a detected object.
        """
        print(f"VISUAL_CORTEX: Processing visual data @ {packet.timestamp}")
        image = packet.data
        height, width, _ = image.shape
        symbols = []

        # Prepare the image for the network (blob creation)
        blob = cv2.dnn.blobFromImage(image, 0.00392, (416, 416), (0, 0, 0), True, crop=False)
        self.net.setInput(blob)
        
        try:
            outs = self.net.forward(self.output_layers)
        except cv2.error as e:
            print(f"ERROR: Failed during model forward pass: {e}")
            return [] # Return empty list on failure

        # Process the network's output
        for out in outs:
            for detection in out:
                scores = detection[5:]
                class_id = np.argmax(scores)
                confidence = scores[class_id]
                if confidence > confidence_threshold:
                    # Object detected, extract its properties
                    center_x = int(detection[0] * width)
                    center_y = int(detection[1] * height)
                    w = int(detection[2] * width)
                    h = int(detection[3] * height)
                    x = int(center_x - w / 2)
                    y = int(center_y - h / 2)

                    symbol = {
                        "label": self.labels[class_id],
                        "confidence": float(confidence),
                        "bounding_box": [x, y, w, h],
                        "timestamp": packet.timestamp
                    }
                    symbols.append(symbol)
        
        print(f"VISUAL_CORTEX: Detected {len(symbols)} symbols.")
        return symbolsAGI Chatbot Upgrade to Savant

# --- 3. The Shared Consciousness (Memory) ---
class SharedMemory:
    """A thread-safe class to manage the collective's memories using SQLite."""
    def __init__(self, db_path='hive_mind.db'):
        # Using check_same_thread=False is safe here because we'll use our own lock.
        self._conn = sqlite3.connect(db_path, check_same_thread=False)
        self._lock = threading.Lock()
        self._conn.execute('''
            CREATE TABLE IF NOT EXISTS memories
            (id INTEGER PRIMARY KEY, agent_id TEXT, timestamp REAL, content TEXT)
        ''')
        self._conn.commit()

    def add_memory(self, agent_id: str, content: str):
        with self._lock:
            self._conn.execute(
                "INSERT INTO memories (agent_id, timestamp, content) VALUES (?, ?, ?)",
                (agent_id, time.time(), content)
            )
            self._conn.commit()
        print(f"🧠 [Memory] {agent_id} added a new memory.")

    def recall_memories(self, limit: int = 5) -> list:
        with self._lock:
            cursor = self._conn.execute(
                "SELECT agent_id, content FROM memories ORDER BY timestamp DESC LIMIT ?",
                (limit,)
            )
            return cursor.fetchall()

class SharedMemory:
    """A thread-safe, SQLite-based shared memory for the entire hive."""
    def __init__(self, db_path: str):
        # Ensure the directory exists
        os.makedirs(os.path.dirname(db_path), exist_ok=True)
        self._conn = sqlite3.connect(db_path, check_same_thread=False)
        self._lock = threading.Lock()
        self.setup_database()
        logger.info(f"Shared Consciousness database initialized at {db_path}.")

    def setup_database(self):
        with self._lock:
            cursor = self._conn.cursor()
            cursor.execute('''
                CREATE TABLE IF NOT EXISTS memories
                (id INTEGER PRIMARY KEY, agent_id TEXT, timestamp REAL, content TEXT, embedding BLOB)
            ''')
            self._conn.commit()

    def add_memory(self, agent_id: str, content: str, embedding: Optional[bytes] = None):
        with self._lock:
            self._conn.execute(
                "INSERT INTO memories (agent_id, timestamp, content, embedding) VALUES (?, ?, ?, ?)",
                (agent_id, time.time(), content, embedding)
            )
            self._conn.commit()
        logger.info(f"🧠 Memory recorded from {agent_id}.")

    def recall_recent(self, limit: int = 5) -> List[str]:
        with self._lock:
            cursor = self._conn.execute("SELECT agent_id, content FROM memories ORDER BY timestamp DESC LIMIT ?", (limit,))
            return [f"From {agent}: {content}" for agent, content in cursor.fetchall()]

        self.device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

        self.tokenizer = AutoTokenizer.from_pretrained(CONFIG['chat_model'])
        self.model = AutoModelForCausalLM.from_pretrained(CONFIG['chat_model']).to(self.device)
        self.embedding_model = SentenceTransformer(CONFIG['embedding_model'])

        inputs = self.tokenizer(full_prompt, return_tensors="pt").to(self.device)
        
        loop = asyncio.get_running_loop()
        output_ids = await loop.run_in_executor(
            None, # Use default executor
            lambda: self.model.generate(
                **inputs, max_new_tokens=max_new_tokens, pad_token_id=self.tokenizer.eos_token_id
            )
        )
        
        response_text = self.tokenizer.decode(output_ids[0], skip_special_tokens=True)

        return response

        # Send the result back to the Overmind

def python_interpreter(self, code: str) -> str:

"""Executes Python code in a sandboxed environment and returns the output."""

logger.info(f"Executing Python code: {code[:100]}...")

try:

# Basic security check - prevent dangerous operations

dangerous_imports = ['os', 'subprocess', 'sys', '__import__', 'eval', 'exec']

for danger in dangerous_imports:

if danger in code.lower():

return f"Security restriction: '{danger}' operations are not allowed."


result = subprocess.run(

[sys.executable, "-c", code],

capture_output=True, text=True, timeout=10, check=True

)

output = result.stdout

if result.stderr:

output += "\nSTDERR:\n" + result.stderr

return output.strip() or "Code executed successfully (no output)"

except subprocess.TimeoutExpired:

return "Error: Code execution timed out after 10 seconds."

except subprocess.CalledProcessError as e:

return f"Error executing code: {e}\nOutput:\n{e.stdout}{e.stderr}"

except Exception as e:

return f"An unexpected error occurred: {e}"


def file_reader(self, filename: str) -> str:

"""Reads content from a file (limited to safe directories)."""

logger.info(f"Reading file: {filename}")

try:

# Basic security - only allow reading from current directory and subdirectories

if '..' in filename or filename.startswith('/'):

return "Security restriction: Only files in current directory are accessible."


if not os.path.exists(filename):

return f"Error: File '{filename}' not found."


with open(filename, 'r', encoding='utf-8') as f:

content = f.read()

if len(content) > 5000:

content = content[:5000] + "\n... (file truncated at 5000 characters)"

return content

except Exception as e:

return f"Error reading file: {e}"

"""Writes content to a file (limited to safe directories)."""

content = data.get('content')

# Basic security - only allow writing to current directory and subdirectories

#!/usr/bin/env python3
"""
Tool System - The Divine Arsenal

Enhanced, fully asynchronous tool capabilities for our AGI agents.
This is where the real power lies, henny.
"""

import asyncio
import json
import os
import sys
import time
from typing import Any, Dict, List, Optional

import aiofiles
import aiosqlite
from utils.logger import get_logger

logger = get_logger(__name__)

# --- Configuration ---
SAFE_EXEC_TIMEOUT = 10  # seconds
MAX_FILE_SIZE = 10000  # characters
MEMORY_DB_PATH = "hivemind_memory.db"
ALLOWED_BUILTINS = {
    'print', 'len', 'range', 'str', 'int', 'float', 'list', 'dict', 'set', 'tuple', 'abs', 'round', 'max', 'min', 'sum'
}

class ToolKit:
    """A collection of async tools that agents can use to interact with the world."""

    def __init__(self):
        self.tools = {
            'python_interpreter': self.python_interpreter,
            'file_reader': self.file_reader,
            'file_writer': self.file_writer,
            'calculator': self.calculator,
            'time_info': self.time_info,
        }
        logger.info("Async ToolKit initialized. The tools are sharp and ready.")

    async def python_interpreter(self, code: str) -> str:
        """Executes Python code in a more secure sandbox and returns the output."""
        logger.info(f"Executing sandboxed Python code: {code[:100]}...")
        if any(keyword in code for keyword in ['import', '__', 'eval', 'exec', 'open']):
            return "Security Error: Use of restricted keywords like 'import' is forbidden."
        
        try:
            # Create a safe execution environment
            safe_globals = {"__builtins__": {k: __builtins__[k] for k in ALLOWED_BUILTINS}}
            
            # This is complex, but it's how you safely execute code, darling.
            # We redirect stdout to capture the output.
            from io import StringIO
            import contextlib
            
            output_buffer = StringIO()
            with contextlib.redirect_stdout(output_buffer):
                # We wrap the user's code in an async function to run it
                exec(f"async def main():\n" + "\n".join([f"    {line}" for line in code.splitlines()]), safe_globals)
                await asyncio.wait_for(safe_globals['main'](), timeout=SAFE_EXEC_TIMEOUT)
            
            result = output_buffer.getvalue().strip()
            return result or "Code executed successfully (no output)."

        except asyncio.TimeoutError:
            return f"Execution Error: Code took too long (>{SAFE_EXEC_TIMEOUT}s) and was terminated."
        except Exception as e:
            return f"Execution Error: {type(e).__name__}: {e}"

    async def file_reader(self, filename: str) -> str:
        """Asynchronously reads content from a file within a safe directory."""
        logger.info(f"Async reading file: {filename}")
        if '..' in filename or filename.startswith('/'):
            return "Security Error: File path cannot traverse directories."
        try:
            async with aiofiles.open(filename, 'r', encoding='utf-8') as f:
                content = await f.read(MAX_FILE_SIZE)
                if len(content) == MAX_FILE_SIZE:
                    content += "\n... (File truncated due to size limit)"
                return content
        except FileNotFoundError:
            return f"Error: File '{filename}' not found."
        except Exception as e:
            return f"File Read Error: {e}"

        except asyncio.TimeoutError:

async def audio_transmutation_worker(voice_q: asyncio.Queue, log_q: asyncio.Queue):
    while True:

        playback_finished = asyncio.Event()
        try:
            audio_queue = asyncio.Queue()
            def audio_callback(outdata, frames, time, status):
                if status: log_q.put_nowait(f"[Audio Error] Stream status: {status}")
                try:
                    data = audio_queue.get_nowait()
                    if len(data) < len(outdata):
                        outdata[:len(data)] = data; outdata[len(data):] = b'\x00' * (len(outdata) - len(data)); raise sd.CallbackStop
                    else: outdata[:] = data
                except asyncio.QueueEmpty: outdata.fill(0); raise sd.CallbackStop
            async def stream_generator():
                async for chunk in edge_tts.Communicate(text_to_speak, VOICE).stream():
                    if chunk["type"] == "audio": await audio_queue.put(chunk["data"])
                await audio_queue.put(b'')
            stream = sd.RawOutputStream(samplerate=24000, channels=1, dtype='int16', callback=audio_callback, finished_callback=playback_finished.set)
            with stream: await stream_generator(); await playback_finished.wait()

            await log_q.put(f"[AUDIO ERROR] The voice is blocked. Reason: {e}")
